            <script>

              const params = new URLSearchParams(window.location.search)

              const goodsId = params.get('goodsId')


              if (goodsId !== null && goodsId !== '') {

                document.getElementById('validContainer').style.display = ''

                openDeepLink()

              } else {

                document.getElementById('invalidContainer').style.display = ''

              }


              function openDeepLink() {

                const config = {

                  iosScheme: 'nateonuc://giftshow-share?product_id=' + goodsId,

                  androidIntent:

                    'intent://giftshow-share?product_id=' + goodsId + '#Intent;scheme=nateonuc;package=Uxpp.UC;end',

                  appStore: 'https://apps.apple.com/kr/app/id362693357',

                  playStore: 'https://play.google.com/store/apps/details?id=Uxpp.UC&hl=ko',

                }


                const userAgent = navigator.userAgent.toLowerCase()

                const isAndroid = /android/i.test(userAgent)

                const isIos = /iphone|ipad|ipod/i.test(userAgent)


                const startTime = Date.now()


                if (isAndroid) {

                  location.href = config.androidIntent


                  setTimeout(() => {

                    if (Date.now() - startTime < 2000) {

                      location.href = config.playStore

                    }

                  }, 1500)

                } else if (isIos) {

                  var iframe = document.createElement('iframe')

                  iframe.style.display = 'none'

                  iframe.src = config.iosScheme

                  document.body.appendChild(iframe)


                  setTimeout(function () {

                    document.body.removeChild(iframe)

                  }, 100)


                  setTimeout(() => {

                    if (Date.now() - startTime < 2000) {

                      location.href = config.appStore

                    }

                  }, 1500)

                } else {

                  alert('모바일 기기에서 접속해 주세요.')

                }

              }

            </script>
